        // Función para verificar si la contraseña ha expirado
        async function checkPasswordExpired(userId) {
            try {
                const userDoc = await db.collection('users').doc(userId).get();
                if (!userDoc.exists) {
                    // Si no existe el documento, crearlo con fecha actual
                    await db.collection('users').doc(userId).set({
                        lastPasswordChange: new Date().toISOString(),
                        passwordHistory: []
                    });
                    return false;
                }
                
                const userData = userDoc.data();
                if (!userData.lastPasswordChange) {
                    // Si no hay fecha de último cambio, establecerla ahora
                    await db.collection('users').doc(userId).update({
                        lastPasswordChange: new Date().toISOString()
                    });
                    return false;
                }
                
                const lastChange = new Date(userData.lastPasswordChange);
                const now = new Date();
                const diffTime = Math.abs(now - lastChange);
                const diffDays = Math.ceil(diffTime / (1000 * 60 * 60 * 24));
                
                return diffDays >= PASSWORD_EXPIRY_DAYS;
            } catch (error) {
                console.error('Error verificando expiración de contraseña:', error);
                return false;
            }
        }
        
        // Función para obtener días restantes antes de expiración
        async function getDaysUntilExpiry(userId) {
            try {
                const userDoc = await db.collection('users').doc(userId).get();
                if (!userDoc.exists) return PASSWORD_EXPIRY_DAYS;
                
                const userData = userDoc.data();
                if (!userData.lastPasswordChange) return PASSWORD_EXPIRY_DAYS;
                
                const lastChange = new Date(userData.lastPasswordChange);
                const now = new Date();
                const diffTime = Math.abs(now - lastChange);
                const diffDays = Math.ceil(diffTime / (1000 * 60 * 60 * 24));
                
                return PASSWORD_EXPIRY_DAYS - diffDays;
            } catch (error) {
                console.error('Error obteniendo días de expiración:', error);
                return PASSWORD_EXPIRY_DAYS;
            }
        }

        // Función para actualizar historial de contraseñas
        async function updatePasswordHistory(userId, newPassword) {
            try {
                const userDoc = await db.collection('users').doc(userId).get();
                let passwordHistory = [];
                
                if (userDoc.exists) {
                    const userData = userDoc.data();
                    passwordHistory = userData.passwordHistory || [];
                }
                
                // Agregar nueva contraseña al historial
                passwordHistory.push(newPassword);
                
                // Mantener solo las últimas PASSWORD_HISTORY_LIMIT contraseñas
                if (passwordHistory.length > PASSWORD_HISTORY_LIMIT) {
                    passwordHistory = passwordHistory.slice(-PASSWORD_HISTORY_LIMIT);
                }
                
                // Actualizar en Firestore
                await db.collection('users').doc(userId).set({
                    lastPasswordChange: new Date().toISOString(),
                    passwordHistory: passwordHistory
                }, { merge: true });
                
            } catch (error) {
                console.error('Error actualizando historial de contraseñas:', error);
            }
        }

                // Mostrar advertencia en la aplicación
                appDaysWarning.style.display = 'block';
                appDaysRemainingText.textContent = `Tu contraseña expira en ${daysRemaining} ${daysRemaining === 1 ? 'día' : 'días'}. Por favor, cámbiala pronto.`;
                
                // También mostrar en el modal de cambio de contraseña si está abierto
                daysRemainingWarning.style.display = 'block';
                daysRemainingText.textContent = `Tu contraseña actual expira en ${daysRemaining} ${daysRemaining === 1 ? 'día' : 'días'}`;
            } else if (daysRemaining <= 0) {
                // Forzar cambio de contraseña
                showChangePasswordModal('Tu contraseña ha expirado. Debes cambiarla para continuar.');
            }
        }

        // Función para guardar obras en Firebase
        async function saveObrasToFirebase() {
            if (!currentUser) return;
            
            try {
                // Guardar cada obra en Firestore
                for (const obra of obras) {
                    // Asegurarse de que la obra tenga el userId
                    obra.userId = currentUser;
                    
                    await db.collection('obras').doc(obra.id).set(obra);
                }
                console.log('Obras guardadas en Firebase');
                
                // También guardar en localStorage como backup
                localStorage.setItem('obras_backup', JSON.stringify(obras));
            } catch (error) {
                console.error('Error guardando en Firebase:', error);
                // Fallback a localStorage
                localStorage.setItem('obras', JSON.stringify(obras));
            }
        }
        
        // Función para cargar obras desde Firebase
        async function loadObrasFromFirebase() {
            if (!currentUser) return;
            
            try {
                const obrasSnapshot = await db.collection('obras')
                    .where('userId', '==', currentUser)
                    .get();
                
                obras = [];
                obrasSnapshot.forEach(doc => {
                    obras.push(doc.data());
                });
                
                console.log(`${obras.length} obras cargadas de Firebase`);
                
                // Si no hay obras en Firebase, intentar cargar del backup
                if (obras.length === 0) {
                    const backup = localStorage.getItem('obras_backup');
                    if (backup) {
                        obras = JSON.parse(backup);
                        // Sincronizar con Firebase
                        await saveObrasToFirebase();
                    }
                }
                
                renderObras();
                return obras;
            } catch (error) {
                console.error('Error cargando de Firebase:', error);
                // Fallback a localStorage
                const localObras = localStorage.getItem('obras');
                obras = localObras ? JSON.parse(localObras) : [];
                renderObras();
                return obras;
            }
        }

        // Función para obtener la fecha actual
        function getCurrentDate() {
            const now = new Date();
            const year = now.getFullYear();
            const month = String(now.getMonth() + 1).padStart(2, '0');
            const day = String(now.getDate()).padStart(2, '0');
            return `${year}-${month}-${day}`;
        }
        
        // Función para formatear fechas
        function formatDate(dateString) {
            if (!dateString) return 'No especificada';
            const date = new Date(dateString);
            return date.toLocaleDateString('es-ES', {
                day: '2-digit',
                month: '2-digit',
                year: 'numeric'
            });
        }
        
        // Función para obtener índices jerárquicos
        function getTaskIndices(obra) {
            const primaryTasks = obra.tareas
                .filter(t => t.tipo === 'primary')
                .sort((a, b) => new Date(a.fechaCreacion) - new Date(b.fechaCreacion));
            
            const secondaryTasks = obra.tareas.filter(t => t.tipo === 'secondary');
            
            const primaryIndices = {};
            primaryTasks.forEach((task, index) => {
                primaryIndices[task.id] = index;
            });
            
            const secondaryIndices = {};
            primaryTasks.forEach(primaryTask => {
                const primaryIndex = primaryIndices[primaryTask.id];
                const secundariasDeEstaPrincipal = secondaryTasks
                    .filter(s => s.parentId === primaryTask.id)
                    .sort((a, b) => new Date(a.fechaCreacion) - new Date(b.fechaCreacion));
                
                secundariasDeEstaPrincipal.forEach((secondaryTask, secondaryIndex) => {
                    secondaryIndices[secondaryTask.id] = `${primaryIndex}.${secondaryIndex + 1}`;
                });
            });
            
            const orphanSecondaries = secondaryTasks.filter(s => !primaryIndices[s.parentId]);
            orphanSecondaries.forEach((task, index) => {
                secondaryIndices[task.id] = `?.${index + 1}`;
            });
            
            return {
                primary: primaryIndices,
                secondary: secondaryIndices,
                getTaskIndex: function(task) {
                    if (task.tipo === 'primary') {
                        return this.primary[task.id] !== undefined ? this.primary[task.id].toString() : '?';
                    } else {
                        return this.secondary[task.id] || '?.?';
                    }
                },
                getDisplayIndex: function(task) {
                    const index = this.getTaskIndex(task);
                    return task.tipo === 'primary' ? `[${index}]` : `[${index}]`;
                }
            };
        }

            taskFormContainer.scrollIntoView({ behavior: 'smooth' });
            document.getElementById('taskNombre').focus();
        }

        // Manejar inicio de sesión - VERSIÓN CORREGIDA
        async function handleLogin(e) {
            e.preventDefault();
            
            if (checkLockout()) return;
            
            const username = document.getElementById('username').value;
            const password = document.getElementById('password').value;
            
            // Verificar primero el nombre de usuario
            if (username !== FIXED_USERNAME) {
                errorMessage.style.display = 'block';
                errorMessage.textContent = 'Usuario incorrecto';
                recordFailedAttempt();
                document.getElementById('password').value = '';
                return;
            }
            
            try {
                // Intentar autenticar con Firebase usando el email fijo
                const userCredential = await auth.signInWithEmailAndPassword(FIXED_EMAIL, password);
                currentUser = userCredential.user.uid;
                localStorage.setItem('currentUser', currentUser);
                localStorage.setItem('username', FIXED_USERNAME);
                
                // Verificar si la contraseña ha expirado
                const passwordExpired = await checkPasswordExpired(currentUser);
                
                if (passwordExpired) {
                    // Mostrar modal de cambio de contraseña obligatorio
                    showChangePasswordModal('Tu contraseña ha expirado. Debes cambiarla para continuar.');
                } else {
                    // Mostrar aplicación principal
                    showApp();
                    errorMessage.style.display = 'none';
                    resetFailedAttempts();
                    
                    // Cargar obras del usuario
                    await loadObrasFromFirebase();
                    
                    // Mostrar advertencia si faltan pocos días para expiración
                    await showPasswordExpiryWarning(currentUser);
                }
                
            } catch (error) {
                // Si hay error de autenticación, verificar si es la contraseña inicial
                if (error.code === 'auth/wrong-password' && password === FIXED_PASSWORD) {
                    // Intentar crear el usuario si no existe, o usar la contraseña inicial
                    try {
                        // Primero intentar crear el usuario si no existe
                        await createFixedUser();
                        
                        // Intentar login con las credenciales iniciales
                        const userCredential = await auth.signInWithEmailAndPassword(FIXED_EMAIL, FIXED_PASSWORD);
                        currentUser = userCredential.user.uid;
                        localStorage.setItem('currentUser', currentUser);
                        localStorage.setItem('username', FIXED_USERNAME);
                        
                        showApp();
                        errorMessage.style.display = 'none';
                        resetFailedAttempts();
                        await loadObrasFromFirebase();
                        await showPasswordExpiryWarning(currentUser);
                        
                    } catch (secondError) {
                        errorMessage.style.display = 'block';
                        errorMessage.textContent = 'Error de autenticación. Contacta al administrador.';
                        recordFailedAttempt();
                    }
                } else if (error.code === 'auth/user-not-found') {
                    // Crear usuario si no existe
                    try {
                        await createFixedUser();
                        
                        // Intentar login nuevamente
                        const userCredential = await auth.signInWithEmailAndPassword(FIXED_EMAIL, password);
                        currentUser = userCredential.user.uid;
                        localStorage.setItem('currentUser', currentUser);
                        localStorage.setItem('username', FIXED_USERNAME);
                        showApp();
                        errorMessage.style.display = 'none';
                        resetFailedAttempts();
                        await loadObrasFromFirebase();
                    } catch (secondError) {
                        errorMessage.style.display = 'block';
                        errorMessage.textContent = 'Error de conexión con el servidor';
                        recordFailedAttempt();
                    }
                } else {
                    errorMessage.style.display = 'block';
                    errorMessage.textContent = 'Contraseña incorrecta';
                    recordFailedAttempt();
                }
                document.getElementById('password').value = '';
            }
        }
        
        // Manejar cambio de contraseña
        async function handleChangePassword(e) {
            e.preventDefault();
            
            const currentPassword = currentPasswordInput.value;
            const newPassword = newPasswordInput.value;
            const confirmPassword = confirmPasswordInput.value;
            
            // Resetear mensajes
            passwordChangeError.style.display = 'none';
            passwordChangeSuccess.style.display = 'none';
            
            // Validaciones básicas
            if (!currentPassword || !newPassword || !confirmPassword) {
                passwordChangeError.textContent = 'Todos los campos son obligatorios';
                passwordChangeError.style.display = 'block';
                return;
            }
            
            if (newPassword !== confirmPassword) {
                passwordChangeError.textContent = 'Las contraseñas no coinciden';
                passwordChangeError.style.display = 'block';
                return;
            }
            
            // Verificar requisitos de contraseña
            const passwordErrors = validatePasswordRequirements(newPassword);
            if (passwordErrors.length > 0) {
                passwordChangeError.innerHTML = passwordErrors.join('<br>');
                passwordChangeError.style.display = 'block';
                return;
            }
            
            // Verificar fortaleza de contraseña
            if (checkPasswordStrength(newPassword) <= 2) {
                passwordChangeError.textContent = 'La contraseña es demasiado débil. Por favor, usa una contraseña más segura.';
                passwordChangeError.style.display = 'block';
                return;
            }
            
            // Verificar que no sea la misma contraseña actual
            if (newPassword === currentPassword) {
                passwordChangeError.textContent = 'La nueva contraseña no puede ser igual a la actual';
                passwordChangeError.style.display = 'block';
                return;
            }
            
            // Verificar que no sea la contraseña fija inicial
            if (newPassword === FIXED_PASSWORD) {
                passwordChangeError.textContent = 'No puedes usar la contraseña inicial del sistema';
                passwordChangeError.style.display = 'block';
                return;
            }
            
            // Verificar historial de contraseñas
            const usedBefore = await checkPasswordHistory(currentUser, newPassword);
            if (usedBefore) {
                passwordChangeError.textContent = 'Esta contraseña ya ha sido usada anteriormente. Por favor, elige una nueva.';
                passwordChangeError.style.display = 'block';
                return;
            }
            
            try {
                const user = auth.currentUser;
                
                // Reautenticar al usuario
                const credential = firebase.auth.EmailAuthProvider.credential(FIXED_EMAIL, currentPassword);
                await user.reauthenticateWithCredential(credential);
                
                // Cambiar la contraseña
                await user.updatePassword(newPassword);
                
                // Actualizar historial de contraseñas
                await updatePasswordHistory(currentUser, newPassword);
                
                // Mostrar mensaje de éxito
                passwordChangeSuccess.textContent = '¡Contraseña cambiada exitosamente! Serás redirigido a la aplicación.';
                passwordChangeSuccess.style.display = 'block';
                
                // Ocultar modal después de 2 segundos
                setTimeout(async () => {
                    hideChangePasswordModal();
                    showApp();
                    
                    // Cargar obras
                    await loadObrasFromFirebase();
                    
                    // Ocultar advertencia de días restantes
                    appDaysWarning.style.display = 'none';
                    
                }, 2000);
                
            } catch (error) {
                passwordChangeError.textContent = 'Error al cambiar la contraseña: ' + 
                    (error.code === 'auth/wrong-password' ? 'Contraseña actual incorrecta' : error.message);
                passwordChangeError.style.display = 'block';
            }
        }

        // Observador de estado de autenticación

                    localStorage.setItem('currentUser', currentUser);
                    localStorage.setItem('username', FIXED_USERNAME);

        // Establecer tipo de tarea
        function setTaskType(type) {
            currentTaskType = type;
            taskTypeInput.value = type;
            
            btnPrimaryTask.classList.toggle('active', type === 'primary');
            btnSecondaryTask.classList.toggle('active', type === 'secondary');
            
            if (type === 'secondary') {
                if (selectedParentTaskId) {
                    parentTaskIndicator.style.display = 'flex';
                    parentTaskContainer.style.display = 'none';
                } else {
                    parentTaskContainer.style.display = 'block';
                    parentTaskIndicator.style.display = 'none';
                }
            } else {
                parentTaskContainer.style.display = 'none';
                parentTaskIndicator.style.display = 'none';
                selectedParentTaskId = null;
            }
            
            if (type === 'secondary' && selectedObraId) {
                loadParentTasks();
            }
            
            if (isEditingTask) {
                const taskId = editingTaskIdInput.value;
                const obra = obras.find(o => o.id === selectedObraId);
                if (obra) {
                    const task = obra.tareas.find(t => t.id === taskId);
                    if (task && task.tipo !== type && type === 'primary') {
                        task.parentId = null;
                    }
                }
            }
        }
        
        // Cargar tareas principales para selector
        function loadParentTasks() {

            if (!obra) return;
            
            const indices = getTaskIndices(obra);
            const primaryTasks = obra.tareas.filter(t => t.tipo === 'primary');
            primaryTasks.sort((a, b) => new Date(a.fechaCreacion) - new Date(b.fechaCreacion));
            
            let options = '<option value="">Selecciona una tarea principal</option>';
            primaryTasks.forEach(task => {
                const index = indices.getTaskIndex(task);
                options += `<option value="${task.id}">[${index}] ${task.nombre}</option>`;
            });
            
            parentTaskSelect.innerHTML = options;
            
            if (selectedParentTaskId) {
                parentTaskSelect.value = selectedParentTaskId;
            }
        }

        // Agregar nueva tarea
        async function addNewTask(obraIndex) {
            const task = {
                id: Date.now().toString(),
                tipo: currentTaskType,
                nombre: document.getElementById('taskNombre').value,
                descripcion: document.getElementById('taskDescripcion').value,
                responsable: document.getElementById('taskResponsable').value,
                fecha: document.getElementById('taskFecha').value || getCurrentDate(),
                completada: false,
                fechaCreacion: new Date().toISOString()
            };
            
            if (currentTaskType === 'secondary') {
                task.parentId = selectedParentTaskId || parentTaskSelect.value;
                if (!task.parentId) {
                    alert('Debes seleccionar una tarea principal para esta tarea secundaria');
                    return;
                }
            }
            
            obras[obraIndex].tareas.push(task);
            await saveObrasToFirebase();
            renderObraDetails();
            
            taskForm.reset();
            selectedParentTaskId = null;
            parentTaskIndicator.style.display = 'none';
            document.getElementById('taskFecha').value = getCurrentDate();
            setTaskType('primary');
            
            if (currentTaskType === 'secondary') {
                loadParentTasks();
            }
        }
        
        // Actualizar tarea existente
        async function updateTask(obraIndex) {
            const taskId = editingTaskIdInput.value;
            const taskIndex = obras[obraIndex].tareas.findIndex(t => t.id === taskId);
            
            if (taskIndex === -1) {
                cancelTaskEdit();
                return;
            }
            
            obras[obraIndex].tareas[taskIndex].tipo = currentTaskType;
            obras[obraIndex].tareas[taskIndex].nombre = document.getElementById('taskNombre').value;
            obras[obraIndex].tareas[taskIndex].descripcion = document.getElementById('taskDescripcion').value;
            obras[obraIndex].tareas[taskIndex].responsable = document.getElementById('taskResponsable').value;
            obras[obraIndex].tareas[taskIndex].fecha = document.getElementById('taskFecha').value;
            
            if (currentTaskType === 'secondary') {
                obras[obraIndex].tareas[taskIndex].parentId = selectedParentTaskId || parentTaskSelect.value;
                if (!obras[obraIndex].tareas[taskIndex].parentId) {
                    alert('Debes seleccionar una tarea principal para esta tarea secundaria');
                    return;
                }
            } else {
                obras[obraIndex].tareas[taskIndex].parentId = null;
            }
            
            await saveObrasToFirebase();
            renderObraDetails();
            cancelTaskEdit();
        }
        
        // Editar una tarea
        function editTask(obraId, taskId) {
            const obra = obras.find(o => o.id === obraId);
            if (!obra) return;
            
            const task = obra.tareas.find(t => t.id === taskId);
            if (!task) return;
            
            isEditingTask = true;
            editingTaskIdInput.value = taskId;
            formTitle.innerHTML = '<i class="fas fa-edit"></i> Editar Tarea';
            editModeBadge.style.display = 'block';
            submitTaskBtn.textContent = 'Actualizar Tarea';
            cancelEditBtn.style.display = 'block';
            
            setTaskType(task.tipo);
            document.getElementById('taskNombre').value = task.nombre;
            document.getElementById('taskDescripcion').value = task.descripcion || '';
            document.getElementById('taskResponsable').value = task.responsable || '';
            document.getElementById('taskFecha').value = task.fecha || getCurrentDate();
            
            if (task.tipo === 'secondary' && task.parentId) {
                selectedParentTaskId = task.parentId;
                loadParentTasks();
                
                const parentTask = obra.tareas.find(t => t.id === task.parentId);
                if (parentTask) {
                    const indices = getTaskIndices(obra);
                    const parentIndex = indices.getTaskIndex(parentTask);
                    parentTaskName.textContent = `[${parentIndex}] ${parentTask.nombre}`;
                    parentTaskIndicator.style.display = 'flex';
                    parentTaskContainer.style.display = 'none';
                }
                
                setTimeout(() => {
                    parentTaskSelect.value = task.parentId;
                }, 100);
            }
            
            taskFormContainer.scrollIntoView({ behavior: 'smooth' });
        }
        
        // Cancelar edición de tarea

            isEditingTask = false;
            editingTaskIdInput.value = '';

        }

        // Eliminar una obra
        async function deleteObra(obraId) {
            if (confirm('¿Estás seguro de que quieres eliminar esta obra? Esta acción no se puede deshacer.')) {
                await deleteObraFromFirebase(obraId);
                obras = obras.filter(o => o.id !== obraId);
                await saveObrasToFirebase();
                renderObras();
                
                if (selectedObraId === obraId) {
                    selectedObraId = null;
                    renderObraDetails();
                }
            }
        }
        
        // Cambiar estado de una tarea
        async function toggleTask(obraId, taskId) {
            const obra = obras.find(o => o.id === obraId);
            if (obra) {
                const task = obra.tareas.find(t => t.id === taskId);
                if (task) {
                    task.completada = !task.completada;
                    await saveObrasToFirebase();
                    renderObraDetails();
                }
            }
        }
        
        // Eliminar una tarea
        async function deleteTask(obraId, taskId) {
            if (!confirm('¿Estás seguro de que quieres eliminar esta tarea?')) return;
            
            const obra = obras.find(o => o.id === obraId);
            if (obra) {
                const taskToDelete = obra.tareas.find(t => t.id === taskId);
                if (taskToDelete && taskToDelete.tipo === 'primary') {
                    obra.tareas = obra.tareas.filter(t => t.parentId !== taskId);
                }
                
                obra.tareas = obra.tareas.filter(t => t.id !== taskId);
                await saveObrasToFirebase();
                renderObraDetails();
                
                if (isEditingTask && editingTaskIdInput.value === taskId) {
                    cancelTaskEdit();
                }
            }
        }

            content += `Sistema de Control de Obras Jerárquico\n`;
            
            // Crear y descargar archivo
            const blob = new Blob([content], { type: 'text/plain;charset=utf-8' });
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = `Resumen_Obra_${obra.nombre.replace(/\s+/g, '_')}_${new Date().toISOString().slice(0, 10)}.txt`;
            document.body.appendChild(a);
            a.click();
            document.body.removeChild(a);
            URL.revokeObjectURL(url);
            
            alert('Informe exportado exitosamente');
        }
        
        // Copiar resumen al portapapeles
        function copySummaryToClipboard() {
            if (!currentSummaryObraId) return;
            
            const obra = obras.find(o => o.id === currentSummaryObraId);
            if (!obra) return;
            
            // Obtener índices jerárquicos
            const indices = getTaskIndices(obra);
            
            // Calcular estadísticas
            const totalTasks = obra.tareas.length;
            const primaryTasks = obra.tareas.filter(t => t.tipo === 'primary');
            const secondaryTasks = obra.tareas.filter(t => t.tipo === 'secondary');
            const completedTasks = obra.tareas.filter(t => t.completada);
            const progress = totalTasks > 0 ? Math.round((completedTasks.length / totalTasks) * 100) : 0;
            
            // Crear contenido del resumen para copiar
            let content = `RESUMEN DE OBRA: ${obra.nombre}\n`;
            content += `Ubicación: ${obra.ubicacion}\n`;
            content += `Cliente: ${obra.cliente}\n`;
            content += `Progreso: ${progress}% (${completedTasks.length}/${totalTasks} tareas completadas)\n\n`;
            
            if (totalTasks > 0) {
                content += `TAREAS (ÍNDICES JERÁRQUICOS):\n`;
                
                // Ordenar tareas principales por fecha de creación
                const primaryTasksSorted = [...primaryTasks].sort((a, b) => 
                    new Date(a.fechaCreacion) - new Date(b.fechaCreacion)
                );
                
                // Agrupar tareas secundarias bajo sus principales
                primaryTasksSorted.forEach(primary => {
                    const primaryIndex = indices.getTaskIndex(primary);
                    const secondaryOfPrimary = secondaryTasks
                        .filter(s => s.parentId === primary.id)
                        .sort((a, b) => new Date(a.fechaCreacion) - new Date(b.fechaCreacion));
                    
                    // Crear línea para tarea principal
                    let line = `• `;
                    if (summaryFieldsConfig.indice) line += `[${primaryIndex}] `;
                    if (summaryFieldsConfig.nombre) line += `${primary.nombre} `;
                    if (summaryFieldsConfig.estado) line += `[${primary.completada ? '✓' : '○'}]`;
                    
                    content += line + '\n';
                    
                    // Mostrar tareas secundarias de esta principal
                    if (secondaryOfPrimary.length > 0) {
                        secondaryOfPrimary.forEach((secondary, secIndex) => {
                            const secondaryIndex = `${primaryIndex}.${secIndex + 1}`;
                            
                            let subLine = `  ◦ `;
                            if (summaryFieldsConfig.indice) subLine += `[${secondaryIndex}] `;
                            if (summaryFieldsConfig.nombre) subLine += `${secondary.nombre} `;
                            if (summaryFieldsConfig.estado) subLine += `[${secondary.completada ? '✓' : '○'}]`;
                            
                            content += subLine + '\n';
                        });
                    }
                });
            }
            
            // Copiar al portapapeles
            navigator.clipboard.writeText(content).then(() => {
                alert('Resumen copiado al portapapeles');
            }).catch(err => {
                console.error('Error al copiar al portapapeles: ', err);
                alert('Error al copiar al portapapeles. Inténtalo de nuevo.');
            });
        }
        
        // Renderizar la lista de obras
        function renderObras() {
            if (obras.length === 0) {
                obrasList.innerHTML = '<div class="empty-message">No hay obras creadas aún. ¡Crea tu primera obra!</div>';
                return;
            }
            
            let html = '';
            
            // Ordenar obras: activas primero, luego completadas
            const obrasOrdenadas = [...obras].sort((a, b) => {
                if (a.estado === 'activa' && b.estado === 'completada') return -1;
                if (a.estado === 'completada' && b.estado === 'activa') return 1;
                return 0;
            });
            
            obrasOrdenadas.forEach(obra => {
                // Calcular estadísticas para la obra
                const totalTasks = obra.tareas.length;
                const completedTasks = obra.tareas.filter(t => t.completada).length;
                const progress = totalTasks > 0 ? Math.round((completedTasks / totalTasks) * 100) : 0;
                
                html += `
                    <div class="obra-item">
                        <div class="obra-header">
                            <div class="obra-title">${obra.nombre}</div>
                            <div class="obra-status ${obra.estado}">${obra.estado === 'activa' ? 'En progreso' : 'Completada'}</div>
                        </div>
                        <div class="obra-description">${obra.descripcion || 'Sin descripción'}</div>
                        <div class="obra-details">
                            <div><i class="fas fa-map-marker-alt"></i> ${obra.ubicacion}</div>
                            <div><i class="fas fa-user-tie"></i> ${obra.cliente}</div>
                            <div><i class="fas fa-calendar-alt"></i> Inicio: ${formatDate(obra.fechaInicio)}</div>
                            <div><i class="fas fa-tasks"></i> ${totalTasks} tareas (${progress}% completado)</div>
                        </div>
                        <div class="obra-actions">
                            <button class="action-btn btn-primary" onclick="selectObra('${obra.id}')">
                                <i class="fas fa-eye"></i> Ver
                            </button>
                            <button class="action-btn btn-info" onclick="showSummary('${obra.id}')">
                                <i class="fas fa-chart-pie"></i> Resumen
                            </button>
                            <button class="action-btn ${obra.estado === 'activa' ? 'btn-success' : 'btn-warning'}" onclick="toggleObraEstado('${obra.id}')">
                                ${obra.estado === 'activa' ? '<i class="fas fa-check"></i> Completar' : '<i class="fas fa-redo"></i> Reactivar'}
                            </button>
                            <button class="action-btn btn-danger" onclick="deleteObra('${obra.id}')">
                                <i class="fas fa-trash"></i> Eliminar
                            </button>
                        </div>
                    </div>
                `;
            });
            
            obrasList.innerHTML = html;
        }
